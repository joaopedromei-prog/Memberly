### Sidebar Navigation Items

```typescript
const navItems = [
  { label: 'Dashboard', href: '/admin', icon: HomeIcon },
  { label: 'Produtos', href: '/admin/products', icon: PackageIcon },
  { label: 'Membros', href: '/admin/members', icon: UsersIcon },
  { label: 'Configurações', href: '/admin/settings', icon: SettingsIcon },
];
```

**Active state detection:** Use `usePathname()` from `next/navigation`. Match exact for Dashboard (`pathname === '/admin'`), startsWith for others (`pathname.startsWith(item.href)` with length > '/admin').

**Icons:** Use simple SVG icons inline or a lightweight icon library. Avoid heavy icon packages. Lucide React (`lucide-react`) is a good lightweight option if needed.

### Zustand UI Store [Source: architecture.md#9-frontend-architecture]

```typescript
// src/stores/ui-store.ts
import { create } from 'zustand';

interface UIState {
  sidebarOpen: boolean;
  toggleSidebar: () => void;
  closeSidebar: () => void;
}

export const useUIStore = create<UIState>((set) => ({
  sidebarOpen: false,
  toggleSidebar: () => set((state) => ({ sidebarOpen: !state.sidebarOpen })),
  closeSidebar: () => set({ sidebarOpen: false }),
}));
```

### Breadcrumb Pattern

```typescript
// src/components/admin/Breadcrumb.tsx
'use client';
import { usePathname } from 'next/navigation';
import Link from 'next/link';

const labelMap: Record<string, string> = {
  admin: 'Dashboard',
  products: 'Produtos',
  members: 'Membros',
  settings: 'Configurações',
  new: 'Novo',
};

export function Breadcrumb() {
  const pathname = usePathname();
  const segments = pathname.split('/').filter(Boolean); // ['admin', 'products', ...]

  return (
    <nav className="flex items-center gap-2 text-sm text-gray-500 mb-6">
      {segments.map((segment, index) => {
        const href = '/' + segments.slice(0, index + 1).join('/');
        const label = labelMap[segment] || segment;
        const isLast = index === segments.length - 1;

        return (
          <span key={href} className="flex items-center gap-2">
            {index > 0 && <span>/</span>}
            {isLast ? (
              <span className="text-gray-900 font-medium">{label}</span>
            ) : (
              <Link href={href} className="hover:text-gray-700">{label}</Link>
            )}
          </span>
        );
      })}
    </nav>
  );
}
```

### Dashboard Stat Card

```typescript
// src/components/admin/StatCard.tsx
import { cn } from '@/lib/utils/cn';

interface StatCardProps {
  title: string;
  value: string | number;
  icon?: React.ReactNode;
  className?: string;
}

export function StatCard({ title, value, icon, className }: StatCardProps) {
  return (
    <div className={cn('bg-white rounded-lg shadow-sm border p-6', className)}>
      <div className="flex items-center justify-between">

          <p className="text-2xl font-bold text-gray-900 mt-1">{value}</p>
        </div>
        {icon && <div className="text-gray-400">{icon}</div>}
      </div>
    </div>
  );
}
```
